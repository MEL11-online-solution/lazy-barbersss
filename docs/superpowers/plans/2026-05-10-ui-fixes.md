## Task 1: Fix light-mode text visibility (globals.css)

**Files:**
- Modify: `frontend/src/styles/globals.css`

**Root cause:** Pages use Tailwind classes like `text-white/60`, `text-white/70` which render as white (with opacity) and are invisible on the light-mode white/gray backgrounds. The CSS vars in globals.css cover body/card/input, but not inline Tailwind opacity-white classes.

**Fix strategy:** Add a CSS attribute-selector override in globals.css that changes `text-white/X` (any opacity-white variant) to `var(--lb-text-muted)` when in light mode AND when inside the `<main>` element. Scoping to `main` naturally excludes the Footer (which always has `bg-navy-900`) and the Navbar (which already uses CSS vars inline). Add a `.keep-white` utility for elements that legitimately need white text on colored backgrounds inside `main` (e.g., barber card gradient headers).

- [ ] **Step 1: Add the light-mode text override to globals.css**

In `frontend/src/styles/globals.css`, add the following block **after** the closing `}` of `@layer components` (before the scrollbar rules):

```css
/* ── Light-mode text visibility fix ─────────────────────────────── */
/* opacity-white Tailwind classes (text-white/40 through text-white/90)
   are invisible on light backgrounds. Override them to use themed muted text
   when inside <main> (excludes Footer which always has dark bg). */
html:not(.dark) main [class*="text-white/"]:not(.keep-white) {
  color: var(--lb-text-muted) !important;
}

/* Utility: force white text regardless of theme (use on colored-bg elements) */
.keep-white {
  color: white !important;
}
```

The final globals.css scrollbar section should look like:

```css

/* ── Scrollbar ───────────────────────────────────────────────────── */
```

```jsx
export default function Chatbot() {
  const [open, setOpen] = useState(false);
  const [messages, setMessages] = useState([
    {
      role: 'bot',
      text: "Hey! 👋 I'm LazyBot. How can I help you today?",
      quickActions: [
        { label: 'Book a cut', action: 'navigate', payload: { path: '/booking' } },
        { label: 'Services', action: 'navigate', payload: { path: '/services' } },
        { label: 'Hours', action: 'intent', payload: { intent: 'hours' } },
        { label: 'Location', action: 'intent', payload: { intent: 'location' } },
      ],
    },
  ]);
  const [input, setInput] = useState('');
  const [busy, setBusy] = useState(false);
  const [, navigate] = useLocation();
  const scrollRef = useRef(null);

  useEffect(() => {
    if (scrollRef.current) {
      scrollRef.current.scrollTop = scrollRef.current.scrollHeight;
    }
  }, [messages, open]);

  async function send(text) {
    const trimmed = (text ?? input).trim();
    if (!trimmed || busy) return;
    setMessages((m) => [...m, { role: 'user', text: trimmed }]);
    setInput('');
    setBusy(true);
    try {
      const r = await chatbotApi.message(trimmed);
      setMessages((m) => [...m, { role: 'bot', text: r.reply, quickActions: r.quick_actions || [] }]);
    } catch (e) {
      setMessages((m) => [...m, { role: 'bot', text: 'Sorry, something went wrong. Try again?' }]);
    } finally {
      setBusy(false);
    }
  }

  function handleQuickAction(qa) {
    if (qa.action === 'navigate' && qa.payload?.path) {
      navigate(qa.payload.path);
      setOpen(false);
    } else if (qa.action === 'intent' && qa.payload?.intent) {
      const map = {
        hours: 'What are your opening hours?',
        location: 'Where are you located?',
        services: 'What services do you offer?',
        prices: 'How much do services cost?',
      };
      send(map[qa.payload.intent] || qa.label);

      navigate('/booking');
      setOpen(false);
    }
  }

        @keyframes lb-pulse {
          0%, 100% { box-shadow: 0 0 0 0 rgba(233,30,99,0.5), 0 0 24px rgba(233,30,99,0.4); }

        .lb-chat-open { animation: lb-slide-up 0.22s ease-out forwards; }
        .lb-dot { animation: lb-dot-bounce 1.2s ease-in-out infinite; }
        .lb-dot:nth-child(2) { animation-delay: 0.15s; }
        .lb-dot:nth-child(3) { animation-delay: 0.3s; }
      `}</style>

      {/* Floating button — shown when closed */}
      {!open && (
        <button
          onClick={() => setOpen(true)}
          className="fixed bottom-6 right-6 z-40 w-14 h-14 rounded-full bg-pink-500 hover:bg-pink-600 text-white text-2xl flex items-center justify-center lb-pulse transition-colors"

                      {m.quickActions.map((qa, idx) => (
                        <button
                          key={idx}
                          onClick={() => handleQuickAction(qa)}
                          className="text-xs px-3 py-1 rounded-full border font-display uppercase tracking-wide transition-colors hover:bg-pink-500 hover:text-white hover:border-pink-500"

                        >
                          {qa.label}
                        </button>
                      ))}
                    </div>
                  )}
                </div>
              </div>
            ))}

            {/* Typing indicator */}
            {busy && (
              <div className="flex justify-start">
                <div
                  className="flex items-center gap-1 px-4 py-3"
                  style={{
                    backgroundColor: 'var(--lb-bg-card)',
                    border: '1px solid var(--lb-border)',
                    borderRadius: '14px',
                  }}
                >
                  <span className="lb-dot w-2 h-2 rounded-full bg-pink-500 inline-block" />
                  <span className="lb-dot w-2 h-2 rounded-full bg-pink-500 inline-block" />
                  <span className="lb-dot w-2 h-2 rounded-full bg-pink-500 inline-block" />
                </div>
              </div>
            )}
          </div>

          {/* Input */}
          <form
            onSubmit={(e) => { e.preventDefault(); send(); }}
            className="flex gap-2 p-3 flex-shrink-0"
            style={{
              borderTop: '1px solid var(--lb-border)',
              backgroundColor: 'var(--lb-bg-card)',
            }}
          >
            <input
              value={input}
              onChange={(e) => setInput(e.target.value)}
              placeholder="Type a message…"
              className="form-input flex-1"
              style={{ borderRadius: '10px' }}
            />
            <button
              type="submit"
              disabled={busy || !input.trim()}
              className="btn-primary btn-sm"
              style={{ borderRadius: '10px', minWidth: '40px' }}
            >
              ➤
            </button>
          </form>
        </div>
      )}
    </>
  );
}
```
